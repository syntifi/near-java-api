10  public class SignatureTest extends AbstractKeyTest {
11      @Test
12      void loadPrivateAndPublicKey_matchesDerivedPublicKey() {
13          assertArrayEquals(alicePublicKey.getKey(), alicePrivateKey.derivePublicKey().getKey());
14          assertArrayEquals(bobPublicKey.getKey(), bobPrivateKey.derivePublicKey().getKey());
15      }
16  
17      @Test
18      void loadPrivateAndPublicKey_Sign_and_Verify_should_be_valid() throws GeneralSecurityException {
19          byte[] aliceMessage = "Hello, Bob!".getBytes();
20          byte[] aliceSignature = alicePrivateKey.sign(aliceMessage);
21  
22          assertTrue(alicePublicKey.verify(aliceMessage, aliceSignature));
23  
24          byte[] bobMessage = "Hello, Alice!".getBytes();
25          byte[] bobSignature = bobPrivateKey.sign(bobMessage);
26  
27          assertTrue(bobPublicKey.verify(bobMessage, bobSignature));
28      }
