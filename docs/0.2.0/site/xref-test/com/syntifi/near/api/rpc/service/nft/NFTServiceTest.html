56      @Test
57      void callContractFunction_NFTContractFunctionCall_forTokensForOwner_return_list() {
58          // Other contracts to test
59          // deadmau55.mintspace2.testnet
60          // bananafratclub.mintspace2.testnet
61          // paras-token-v2.testnet
62          NFTContract contract = new NFTContract("paras-token-v2.testnet");
63  
64          contract.setMetadata(service.getMetadata(nearClient, contract.getContractId()));
65  
66          FunctionCallResult<NFTTokenList> tokenList = service.getTokensForOwner(nearClient,
67                  contract.getContractId(), "wallet-test.testnet", "1");
68  
69          LOGGER.debug("{}", tokenList.getContractFunctionCallResult().getResult());
70          for (NFTToken token : tokenList.getResult()) {
71              LOGGER.debug("Token: {}", token);
72              assertNotNull(token);
73              assertNotNull(token.getTokenId());
74  
75              assertDoesNotThrow(() -> token.getMediaOrReferenceURL(contract));
76              NFTTokenMediaURL nftTokenMediaURL = token.getMediaOrReferenceURL(contract);
77              LOGGER.debug("Token Media: {} of type {}", nftTokenMediaURL.getUrl().toString(), nftTokenMediaURL.getType().toString());
78              if (nftTokenMediaURL.getUrl() == null) {
79                  assertEquals(NFTTokenMediaURL.Type.EMPTY, nftTokenMediaURL.getType());
80              } else {
81                  assertNotEquals(NFTTokenMediaURL.Type.EMPTY, nftTokenMediaURL.getType());
82              }
83          }
84      }
