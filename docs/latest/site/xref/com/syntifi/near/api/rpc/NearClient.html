51      /**
52       * NearService builder
53       *
54       * @param network the rpc peer url to connect to
55       * @return the data holding object
56       * @throws NearException thrown when url is invalid or unparseable
57       */
58      static NearClient usingNetwork(Network network) throws NearException {
59          Map<String, String> customHeaders = new TreeMap<>();
60          customHeaders.put("Content-Type", "application/json");
61  
62          try {
63              JsonRpcHttpClient client = new JsonRpcHttpClient(NearRpcObjectMapper.INSTANCE,
64                      new URL("https://" + network.getRpcUrl()),
65                      customHeaders);
66  
67              client.setExceptionResolver(new NearExceptionResolver());
68  
69              return ProxyUtil.createClientProxy(NearClient.class.getClassLoader(), NearClient.class, client);
70          } catch (MalformedURLException e) {
71              throw new NearException("Invalid URL " + network.getRpcUrl(), e.getCause());
72          }
73      }
