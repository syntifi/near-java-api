19  public class BaseService {
20      /**
21       * Prepares the transaction to send with a given nonce
22       *
23       * @param nearService      the near service instance to use
24       * @param signerId         the signer id
25       * @param receiverId       the receiver id
26       * @param signerPublicKey  signer public key
27       * @param signerPrivateKey signer private key
28       * @param actionList       list of {@link Action} to send
29       * @param nonce            nonce to add in the transaction
30       * @return the base64 encoded signed transaction string
31       * @throws GeneralSecurityException thrown if failed to sign the transaction
32       */
33      public static String prepareTransactionForActionList(NearService nearService, String signerId, String receiverId,
34                                                           PublicKey signerPublicKey, PrivateKey signerPrivateKey,
35                                                           List<Action> actionList, Long nonce)
36              throws GeneralSecurityException {
37          Block block = nearService.getBlock(Finality.FINAL);
38  
39          Transaction transaction = Transaction
40                  .builder()
41                  .signerId(signerId)
42                  .publicKey(signerPublicKey)
43                  .nonce(nonce)
44                  .receiverId(receiverId)
45                  .blockHash(block.getHeader().getHash().getDecodedHash())
46                  .actions(actionList)
47                  .build();
48  
49          byte[] serializedTx = Borsh.serialize(transaction);
50          byte[] hashedTx = Sha256.digest(serializedTx);
51          byte[] signedTx = signerPrivateKey.getPrivateKey().sign(hashedTx);
52  
53          SignedTransaction signedTransaction =
54                  SignedTransaction.builder()
55                          .transaction(transaction)
56                          .signature(Signature.builder()
57                                  .keyType(signerPublicKey.getType())
58                                  .data(signedTx).build())
59                          .build();
60  
61          byte[] borshTx = Borsh.serialize(signedTransaction);
62  
63          return Base64.getEncoder().encodeToString(borshTx);
64      }

67      /**
68       * Prepares the transaction to send and automatically queries for the next nonce
69       *
70       * @param nearService      the near service instance to use
71       * @param signerId         the signer id
72       * @param receiverId       the receiver id
73       * @param signerPublicKey  signer public key
74       * @param signerPrivateKey signer private key
75       * @param actionList       list of {@link Action} to send
76       * @return the base64 encoded signed transaction string
77       * @throws GeneralSecurityException thrown if failed to sign the transaction
78       */
79      public static String prepareTransactionForActionList(NearService nearService, String signerId, String receiverId,
80                                                           PublicKey signerPublicKey, PrivateKey signerPrivateKey,
81                                                           List<Action> actionList) throws GeneralSecurityException {
82         AccessKey accessKey = nearService.viewAccessKey(Finality.FINAL, signerId, signerPublicKey.toEncodedBase58String());
83         return BaseService.prepareTransactionForActionList(nearService, signerId, receiverId,
84                 signerPublicKey, signerPrivateKey, actionList,accessKey.getNonce() + 1L);
85      }
