11  /**
12   * Caller for contract methods
13   *
14   * @author Alexandre Carvalho
15   * @author Andre Bertolace
16   * @since 0.2.0
17   */
18  public abstract class ContractMethodCaller {
19  
20      /**
21       * Calls a contract for specific method
22       *
23       * @param method            the method to call
24       * @param returnClass       the type of result object
25       * @param nearService       the near service to use
26       * @param contractAccountId the contract account id
27       * @param params            the method parameters
28       * @param <T>               type of the result object
29       * @return a typed FunctionCallResult for the requested contract method
30       * @throws IOException thrown when fails to map json to result object
31       */
32      public static <T> FunctionCallResult<T> callFor(ContractMethod method, Class<T> returnClass, NearService nearService, String contractAccountId, ContractMethodParams params) throws IOException {
33          ContractFunctionCallResult contractFunctionCallResult = FunctionCall.builder()
34                  .finality(Finality.OPTIMISTIC)
35                  .methodName(method.getMethodName())
36                  .accountId(contractAccountId)
37                  .args(params != null ? params.toJsonBase64String() : Base64String.fromDecodedString(""))
38                  .build()
39                  .call(nearService);
40          return new FunctionCallResult<>(contractFunctionCallResult, returnClass);
41      }
42  }
