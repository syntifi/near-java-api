3   import java.io.ByteArrayOutputStream;
4   import java.io.FileInputStream;
5   import java.io.IOException;
6   import java.io.InputStream;
7   
8   import com.fasterxml.jackson.core.JsonProcessingException;
9   import com.fasterxml.jackson.databind.JsonMappingException;
10  import com.fasterxml.jackson.databind.ObjectMapper;
11  import com.syntifi.near.api.service.NearObjectMapper;
12  
13  /**
14   * Basic support functionality for test case classes
15   * 
16   * @author Alexandre Carvalho
17   * @author Andre Bertolace
18   * @since 0.0.1
19   */
20  public final class JsonHelper {
21      public static final ObjectMapper OBJECT_MAPPER = new NearObjectMapper();
22  
23      /**
24       * Loads test json from resources
25       *
26       * @param filename filename to load
27       * @return file content
28       * @throws IOException thrown if error reading file
29       */
30      public static String loadJsonFromFile(String filename) throws IOException {
31          String fileJson;
32          ByteArrayOutputStream baos = new ByteArrayOutputStream();
33  
34          try (InputStream is = new FileInputStream(filename)) {
35              // copy stream
36              byte[] buffer = new byte[1024];
37              int bytesRead;
38              while ((bytesRead = is.read(buffer)) != -1) {
39                  baos.write(buffer, 0, bytesRead);
40              }
41  
42              fileJson = baos.toString();
43          }
44          return fileJson;
45      }
46  
47      /**
48       * Loads test json from resources
49       *
50       * @param filename filename to load
51       * @return file content
52       * @throws IOException thrown if error reading file
53       */
54      public static String loadJsonFromResourceFile(String filename) throws IOException {
55          String fileJson;
56          ByteArrayOutputStream baos = new ByteArrayOutputStream();
57  
58          try (InputStream is = JsonHelper.class.getClassLoader().getResourceAsStream(filename)) {
59              // copy stream
60              byte[] buffer = new byte[1024];
61              int bytesRead;
62              while (true) {
63                  assert is != null;
64                  if ((bytesRead = is.read(buffer)) == -1) break;
65                  baos.write(buffer, 0, bytesRead);
66              }
67  
68              fileJson = baos.toString();
69          }
70          return fileJson;
71      }
72  
73      /**
74       * Prettifies json for assertion consistency
75       * 
76       * @param json json string to prettify
77       * @return prettified json
78       * @throws JsonMappingException thrown if error mapping json
79       * @throws JsonProcessingException thrown if error processing json
80       */
81      public static String getPrettyJson(String json) throws JsonMappingException, JsonProcessingException {
82          Object jsonObject = OBJECT_MAPPER.readValue(json, Object.class);
83          return getPrettyJson(jsonObject);
84      }
85  
86      /**
87       * Prettifies json for assertion consistency
88       * 
89       * @param jsonObject object to serialize and prettify
90       * @return prettified json
91       * @throws JsonMappingException thrown if error mapping json
92       * @throws JsonProcessingException thrown if error processing json
93       */
94      public static String getPrettyJson(Object jsonObject) throws JsonProcessingException {
95          return OBJECT_MAPPER.writerWithDefaultPrettyPrinter().writeValueAsString(jsonObject);
96      }
97  }
