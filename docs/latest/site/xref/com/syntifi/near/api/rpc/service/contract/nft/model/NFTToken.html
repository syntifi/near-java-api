22  @Getter
23  @Setter
24  @ToString
25  @JsonIgnoreProperties(value = {"approved_account_ids"})
26  public class NFTToken {
27  
28      private static final String DEFAULT_MEDIA_URL = "https://cloudflare-ipfs.com/ipfs/%s";
29  
30      @JsonProperty("token_id")
31      private String tokenId;
32      @JsonProperty("owner_id")
33      private String ownerId;
34  
35      private NFTTokenMetadata metadata;
36  
37      @JsonIgnore
38      public String getMediaUrl(NFTContract contract) throws URISyntaxException, MalformedURLException {
39          URI uri;
40          if (metadata.getMedia() == null) {
41              uri = null; // no url if no media
42          } else if (Strings.isURL(metadata.getMedia()) || metadata.getMedia().startsWith("data:image")) {
43              uri = new URI(metadata.getMedia());
44          } else if (contract.getMetadata().getResult().getBaseUri() != null) {
45              uri = new URI(contract.getMetadata().getResult().getBaseUri() + "/" + metadata.getMedia());
46          } else {
47              uri = new URI(String.format(DEFAULT_MEDIA_URL, metadata.getMedia()));
48          }
49  
50          return uri == null ? null : uri.normalize().toURL().toString();
51      }
52  }
