37      /**
38       * Get the token media or media reference url given a contract
39       *
40       * You can parse correctly to an image or reference json by checking the response content-type
41       *
42       * 1 - First no media and no reference - no url nor object - returns null
43       * 2 - has media (ignore reference in this case) - returns String with url
44       * 3 - has reference (only considered if media is null) - returns String with url for off-chain data JSON
45       *
46       * @param contract the token contract
47       * @return the media url
48       * @throws NearException if fails to parse the url
49       */
50      @JsonIgnore
51      public NFTTokenMediaURL getMediaOrReferenceURL(NFTContract contract) throws NearException {
52          NFTTokenMediaURL returnURL = new NFTTokenMediaURL();
53  
54          try {
55              URI uri;
56              String baseUri = contract.getMetadata().getResult().getBaseUri();
57  
58              if (metadata.getMedia() != null && (Strings.isURL(metadata.getMedia()) || metadata.getMedia().startsWith("data:image"))) {
59                  uri = new URI((baseUri != null ? baseUri + "/" : "") + metadata.getMedia());
60                  returnURL.setType(NFTTokenMediaURL.Type.MEDIA);
61              } else if (metadata.getMedia() != null) {
62                  uri = new URI((baseUri != null ? baseUri + "/" : "") + metadata.getMedia());
63                  returnURL.setType(NFTTokenMediaURL.Type.MEDIA);
64              } else if (metadata.getReference() != null) {
65                  uri = new URI((baseUri != null ? baseUri + "/" : "") + metadata.getReference());
66                  returnURL.setType(NFTTokenMediaURL.Type.REFERENCE);
67              } else {
68                  uri = null;
69                  returnURL.setType(NFTTokenMediaURL.Type.EMPTY);
70              }
71  
72              returnURL.setUrl((uri == null) ? null : uri.normalize().toURL());
73          } catch (MalformedURLException | URISyntaxException e) {
74              throw new NearException("Invalid url for media/reference", e);
75          }
76  
77          return returnURL;
78      }
79  }
