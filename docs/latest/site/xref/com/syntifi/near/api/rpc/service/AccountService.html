30      /**
31       * Sends a {@link TransferAction} transaction calling a contract to create_account
32       * and waits for result using  {@link NearService#sendTransactionAwait(String)}
33       *
34       * @param nearService          the near service instance to use
35       * @param newAccountId         human-readable id of new account
36       * @param newAccountPublicKey  public key of new account
37       * @param amountToNewAccount   the amount to transfer
38       * @param creatorAccountId     human-readable id of creatot account
39       * @param creatorPublicKey     signer/account creator public key
40       * @param creatorPrivateKey    signer/account creator privatekey
41       * @return {@link TransactionAwait} object with the result
42       * @throws GeneralSecurityException thrown if failed to sign the transaction
43       * @throws NearServiceException any NEAR-RPC exception
44       */
45      public static TransactionAwait createNamedAccount(NearService nearService, String newAccountId,
46                                                        PublicKey newAccountPublicKey, BigInteger amountToNewAccount,
47                                                        String creatorAccountId, PublicKey creatorPublicKey,
48                                                        PrivateKey creatorPrivateKey)
49              throws GeneralSecurityException, NearServiceException {
50          long nonce = 0L;
51          ObjectMapper mapper = new ObjectMapper();
52          ObjectNode args = mapper.createObjectNode();
53          args.put("new_account_id", newAccountId);
54          args.put("new_public_key", newAccountPublicKey.getJsonPublicKey());
55  
56          List<Action> actions = Arrays.asList(FunctionCallAction.builder()
57                  .methodName("create_account")
58                  .args(args.toString())
59                  .gas(BigInteger.valueOf(300000000000000L))
60                  .deposit(amountToNewAccount == null ? "" : amountToNewAccount.toString())
61                  .build());
62          //TODO: receiverId can be either testnet or near (depending on the network)
63          return nearService.sendTransactionAwait(BaseService.prepareTransactionForActionList(
64                  nearService, creatorAccountId, "testnet", creatorPublicKey, creatorPrivateKey,
65                  actions, nonce));
66      }

68      /**
69       * Internal method to create a list of actions needed to generate a new subaccount
70       *
71       * @param nearService          the near service instance to use
72       * @param amountToNewAccount   the amount to transfer
73       * @param creatorAccountId     human-readable id of creatot account
74       * @param creatorPublicKey     signer/account creator public key
75       * @return List of actions
76       */
77      private static List<Action> createActionArrayToCreateSubAccount(NearService nearService, BigInteger amountToNewAccount,
78                                                                      String creatorAccountId, PublicKey creatorPublicKey) {
79          AccessKey accessKey = nearService.viewAccessKey(Finality.FINAL, creatorAccountId, creatorPublicKey.toEncodedBase58String());
80          long nonce = accessKey.getNonce();
81          Action[] actionsArray = new Action[]{CreateAccountAction.builder()
82                  .build(),
83                  TransferAction.builder()
84                          .deposit(amountToNewAccount)
85                          .build(),
86                  AddKeyAction.builder()
87                          .publicKey(creatorPublicKey)
88                          .accessKey(AccessKey.builder()
89                                  .nonce(nonce)
90                                  .permission(FullAccessPermission.FULL_ACCESS)
91                                  .build())
92                          .build()};
93          return Arrays.asList(actionsArray);
94      }

96      /**
97       * Sends a {@link TransferAction} transaction with the needed actions to generate a sub account
98       * and waits for result using  {@link NearService#sendTransactionAwait(String)}
99       *
100      * @param nearService          the near service instance to use
101      * @param newAccountId         human-readable id of new account
102      * @param amountToNewAccount   the amount to transfer
103      * @param creatorAccountId     human-readable id of creatot account
104      * @param creatorPublicKey     signer/account creator public key
105      * @param creatorPrivateKey    signer/account creator privatekey
106      * @return {@link TransactionAwait} object with the result
107      * @throws GeneralSecurityException thrown if failed to sign the transaction
108      */
109     public static TransactionAwait createSubAccount(NearService nearService, String newAccountId,
110                                                     BigInteger amountToNewAccount, String creatorAccountId,
111                                                     PublicKey creatorPublicKey, PrivateKey creatorPrivateKey)
112             throws GeneralSecurityException {
113         List<Action> actions = AccountService.createActionArrayToCreateSubAccount(nearService, amountToNewAccount,
114                 creatorAccountId, creatorPublicKey);
115 
116         return nearService.sendTransactionAwait(BaseService.prepareTransactionForActionList(
117                 nearService, creatorAccountId, newAccountId, creatorPublicKey, creatorPrivateKey,
118                 actions));
119     }

121     /**
122      * Sends a {@link TransferAction} transaction async using {@link NearService#sendTransactionAsync(String)}
123      * with the needed actions to generate a sub account and waits for result using
124      * {@link NearService#sendTransactionAwait(String)}
125      *
126      * @param nearService          the near service instance to use
127      * @param newAccountId         human-readable id of new account
128      * @param amountToNewAccount   the amount to transfer
129      * @param creatorAccountId     human-readable id of creatot account
130      * @param creatorPublicKey     signer/account creator public key
131      * @param creatorPrivateKey    signer/account creator privatekey
132      * @return transatcion hash
133      * @throws GeneralSecurityException thrown if failed to sign the transaction
134      */
135     public static EncodedHash createSubAccountAsync(NearService nearService, String newAccountId,
136                                                     BigInteger amountToNewAccount, String creatorAccountId,
137                                                     PublicKey creatorPublicKey, PrivateKey creatorPrivateKey) throws GeneralSecurityException {
138         List<Action> actions = AccountService.createActionArrayToCreateSubAccount(nearService, amountToNewAccount,
139                 creatorAccountId, creatorPublicKey);
140         return EncodedHash.builder()
141                 .encodedHash(nearService.sendTransactionAsync(BaseService.prepareTransactionForActionList(
142                         nearService, creatorAccountId, newAccountId, creatorPublicKey, creatorPrivateKey,
143                         actions)))
144                 .build();
145     }

147     /**
148      * Method to check that the human-readable account id follows the naming convention
149      *
150      * @param accountId human-readable account id
151      * @throws NearServiceException thows if the name deviates from the regex pattern
152      */
153     public static void checkAccountId(String accountId) throws NearServiceException {
154         Pattern regex = Pattern.compile("^(([a-z\\d]+[-_])*[a-z\\d]+.)*([a-z\\d]+[-_])*[a-z\\d]+$");
155         Matcher matcher = regex.matcher(accountId);
156         if (matcher.groupCount() == 0 || matcher.group(0).length() < 2 || accountId.length() > 64) {
157             throw new NearServiceException("accountId does not follow the standard: minimum length is 2, " +
158                     "maximum length is 64, accountId consists of accountId parts separated by . ," +
159                     "and accountId part consists of lowercase alphanumeric symbols separated by either _ or - ",
160                     null);
161         }
162         return;
163     }
