14  /**
15   * KeySig holds information about a key or signature.
16   * Also, it handles to disassemble and assemble of key and signature structure.
17   *
18   * @author Alexandre Carvalho
19   * @author Andre Bertolace
20   * @since 0.0.1
21   */
22  @Getter
23  @Setter
24  @NoArgsConstructor
25  @EqualsAndHashCode
26  public class KeySig {
27  
28      private static final String SEPARATOR = ":";
29  
30      @BorshField(order = 1)
31      @JsonIgnore
32      protected KeyType type;
33  
34      @BorshField(order = 2)
35      @JsonIgnore
36      protected byte[] data;
37  
38      public KeySig(KeyType keyType, byte[] data) {
39          this.type = keyType;
40          this.data = data;
41      }
42  
43      /**
44       * Reads from key or signature from base-58 string
45       *
46       * @param encodedBase58String a string base-58 encoded with the key or signature
47       * @param clazz               return class type
48       * @param <T>                 the parameter return class
49       * @return a PublicKey, PrivateKey or Signature object with type and data
50       */
51      public static <T extends KeySig> T fromEncodedBase58String(String encodedBase58String, Class<T> clazz) {
52          KeyType keyType;
53          byte[] data;
54  
55          String[] key = encodedBase58String.split(SEPARATOR, -1);
56          try {
57              keyType = KeyType.valueOf(key[0].toUpperCase());
58          } catch (Exception e) {
59              throw new NoSuchTypeException(String.format("No implementation found for key type %s", key[0]), e);
60          }
61  
62          data = Base58.decode(key[1]);
63  
64          try {
65              return clazz.getDeclaredConstructor(KeyType.class, byte[].class).newInstance(keyType, data);
66          } catch (InvocationTargetException | InstantiationException | IllegalAccessException | NoSuchMethodException e) {
67              throw new NoSuchTypeException(String.format("Class %s instantiation failed for needed constructor", clazz.getSimpleName()), e);
68          }
69      }
70  
71      public String toEncodedBase58String() {
72          return this.getType().toString().toLowerCase() + SEPARATOR + this;
73      }
74  
75      @Override
76      public String toString() {
77          return Base58.encode(data);
78      }
79  }
