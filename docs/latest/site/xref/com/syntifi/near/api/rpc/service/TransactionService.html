22  /**
23   * Base service provides methods to sign transactions
24   *
25   * @author Alexandre Carvalho
26   * @author Andre Bertolace
27   * @since 0.2.0
28   */
29  @NoArgsConstructor(access = AccessLevel.PRIVATE)
30  public class TransactionService {
31      /**
32       * Prepares the transaction to send with a given nonce
33       *
34       * @param nearClient      the near service instance to use
35       * @param signerId         the signer id
36       * @param receiverId       the receiver id
37       * @param signerPublicKey  signer public key
38       * @param signerPrivateKey signer private key
39       * @param actionList       list of {@link Action} to send
40       * @param nonce            nonce to add in the transaction
41       * @return the base64 encoded signed transaction string
42       * @throws GeneralSecurityException thrown if failed to sign the transaction
43       */
44      public static String prepareTransactionForActionList(NearClient nearClient, String signerId, String receiverId,
45                                                           PublicKey signerPublicKey, PrivateKey signerPrivateKey,
46                                                           List<Action> actionList, Long nonce)
47              throws GeneralSecurityException {
48          Block block = nearClient.getBlock(Finality.FINAL);
49  
50          Transaction transaction = Transaction
51                  .builder()
52                  .signerId(signerId)
53                  .publicKey(signerPublicKey)
54                  .nonce(nonce)
55                  .receiverId(receiverId)
56                  .blockHash(block.getHeader().getHash().getDecodedHash())
57                  .actions(actionList)
58                  .build();
59  
60          byte[] serializedTx = Borsh.serialize(transaction);
61          byte[] hashedTx = Sha256.digest(serializedTx);
62          byte[] signedTx = signerPrivateKey.getPrivateKey().sign(hashedTx);
63  
64          SignedTransaction signedTransaction =
65                  SignedTransaction.builder()
66                          .transaction(transaction)
67                          .signature(Signature.builder()
68                                  .keyType(signerPublicKey.getType())
69                                  .data(signedTx).build())
70                          .build();
71  
72          byte[] borshTx = Borsh.serialize(signedTransaction);
73  
74          return Base64.getEncoder().encodeToString(borshTx);
75      }

78      /**
79       * Prepares the transaction to send and automatically queries for the next nonce
80       *
81       * @param nearClient      the near service instance to use
82       * @param signerId         the signer id
83       * @param receiverId       the receiver id
84       * @param signerPublicKey  signer public key
85       * @param signerPrivateKey signer private key
86       * @param actionList       list of {@link Action} to send
87       * @return the base64 encoded signed transaction string
88       * @throws GeneralSecurityException thrown if failed to sign the transaction
89       */
90      public static String prepareTransactionForActionList(NearClient nearClient, String signerId, String receiverId,
91                                                           PublicKey signerPublicKey, PrivateKey signerPrivateKey,
92                                                           List<Action> actionList) throws GeneralSecurityException {
93         AccessKey accessKey = nearClient.viewAccessKey(Finality.FINAL, signerId, signerPublicKey.toEncodedBase58String());
94         return TransactionService.prepareTransactionForActionList(nearClient, signerId, receiverId,
95                 signerPublicKey, signerPrivateKey, actionList,accessKey.getNonce() + 1L);
96      }
97  }
