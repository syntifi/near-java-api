19  /**
20   * Serializer to handle the polymorphic terms with subtype annotations
21   *
22   * @author Alexandre Carvalho
23   * @author Andre Bertolace
24   * @since 0.3.0
25   */
26  public class SubTypesDeserializer<T> extends JsonDeserializer<T> {
27      private final Map<String, Class<?>> propertyNameToType;
28  
29      public SubTypesDeserializer(Class<T> typeParameterClass)  {
30          this.propertyNameToType = Arrays.stream(typeParameterClass.getAnnotation(JsonSubTypes.class).value())
31                  .collect(Collectors.toMap(JsonSubTypes.Type::name, JsonSubTypes.Type::value,
32                          (a, b) -> a, HashMap::new));
33      }
34  
35      @Override
36      @SuppressWarnings("unchecked")
37      public T deserialize(JsonParser jsonParser, DeserializationContext deserializationContext) throws IOException {
38          ObjectMapper objectMapper = (ObjectMapper) jsonParser.getCodec();
39          ObjectNode object = objectMapper.readTree(jsonParser);
40          for (Iterator<String> it = object.fieldNames(); it.hasNext(); ) {
41              String field = it.next();
42              if (propertyNameToType.containsKey(field)) {
43                  return (T) objectMapper.treeToValue(object, propertyNameToType.get(field));
44              }
45          }
46          throw new IllegalArgumentException("could not infer to which class to deserialize " + object);
47      }
48  
49  }
