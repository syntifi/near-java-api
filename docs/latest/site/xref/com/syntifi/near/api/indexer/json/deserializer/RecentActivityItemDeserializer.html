24  /**
25   * Custom serializer for a recent activity item
26   *
27   * @author Alexandre Carvalho
28   * @author Andre Bertolace
29   * @since 0.2.0
30   */
31  public class RecentActivityItemDeserializer extends JsonDeserializer<RecentActivityItem> {
32      public RecentActivityItem deserialize(JsonParser p, DeserializationContext ctxt)
33              throws IOException {
34          JsonNode node = p.getCodec().readTree(p);
35  
36          final RecentActivityItem recentActivityItem = RecentActivityItem.builder()
37                  .blockHash(node.get("block_hash").asText())
38                  .blockTimestamp(node.get("block_timestamp").asText())
39                  .hash(node.get("hash").asText())
40                  .actionIndex(node.get("action_index").asInt())
41                  .signerId(node.get("signer_id").asText())
42                  .receiverId(node.get("receiver_id").asText())
43                  .actionKind(RecentActivityItem.RecentActivityActionKind.valueOf(node.get("action_kind").asText()))
44                  .build();
45  
46          JsonNode argsNode = node.get("args");
47          if (RecentActivityItem.RecentActivityActionKind.CREATE_ACCOUNT.equals(recentActivityItem.getActionKind())) {
48              recentActivityItem.setArgs(RecentActivityArgCreateAccount.builder().build());
49          } else if (RecentActivityItem.RecentActivityActionKind.DEPLOY_CONTRACT.equals(recentActivityItem.getActionKind())) {
50              recentActivityItem.setArgs(RecentActivityArgDeployContract.builder()
51                      .code(argsNode.get("code").binaryValue()).build());
52          } else if (RecentActivityItem.RecentActivityActionKind.FUNCTION_CALL.equals(recentActivityItem.getActionKind())) {
53              recentActivityItem.setArgs(RecentActivityArgFunctionCall.builder()
54                      .methodName(argsNode.get("method_name").asText())
55                      .argsJson(argsNode.get("args_json").asText())
56                      .argsBase64(argsNode.get("args_base64").asText())
57                      .gas(argsNode.get("gas").asLong())
58                      .deposit(argsNode.get("deposit").asText())
59                      .build());
60          } else if (RecentActivityItem.RecentActivityActionKind.TRANSFER.equals(recentActivityItem.getActionKind())) {
61              recentActivityItem.setArgs(
62                      RecentActivityArgTransfer.builder()
63                              .deposit(argsNode.get("deposit").asText())
64                              .build());
65          } else if (RecentActivityItem.RecentActivityActionKind.STAKE.equals(recentActivityItem.getActionKind())) {
66              recentActivityItem.setArgs(RecentActivityArgStake.builder()
67                      .stake(new BigInteger(argsNode.get("stake").asText()))
68                      .publicKey(PublicKey.getPublicKeyFromJson(argsNode.get("public_key").asText()))
69                      .build());
70          } else if (RecentActivityItem.RecentActivityActionKind.ADD_KEY.equals(recentActivityItem.getActionKind())) {
71              JsonNode accessKey = argsNode.get("access_key");
72              recentActivityItem.setArgs(
73                      RecentActivityArgAddKey.builder()
74                              .accessKey(
75                                      RecentActivityAccessKey.builder()
76                                              .nonce(accessKey.get("nonce").asLong())
77                                              .permission(RecentActivityPermission.builder()
78                                                      .permissionKind(
79                                                              RecentActivityPermission.PermissionKind.valueOf(accessKey.get("permission").get("permission_kind").asText()))
80                                                      .build())
81                                              .build())
82                              .publicKey(argsNode.get("public_key").asText())
83                              .build());
84          } else if (RecentActivityItem.RecentActivityActionKind.DELETE_KEY.equals(recentActivityItem.getActionKind())) {
85              recentActivityItem.setArgs(RecentActivityArgDeleteKey.builder()
86                      .publicKey(PublicKey.getPublicKeyFromJson(argsNode.get("public_key").asText()))
87                      .build());
88          } else if (RecentActivityItem.RecentActivityActionKind.DELETE_ACCOUNT.equals(recentActivityItem.getActionKind())) {
89              recentActivityItem.setArgs(RecentActivityArgDeleteAccount.builder()
90                      .beneficiaryId(argsNode.get("beneficiary_id").asText())
91                      .build());
92          } else {
93              throw new NearIndexerException(
94                      String.format("Action %s not mapped for deserialization", recentActivityItem.getActionKind()));
95          }
96  
97          return recentActivityItem;
98      }
99  }
