14  /**
15   * The interface for contract method clients
16   *
17   * @author Alexandre Carvalho
18   * @author Andre Bertolace
19   * @since 0.2.0
20   */
21  public interface ContractMethodProxy {
22      /**
23       * Proxy invoke the contracts view methods by sending the Base64String
24       *
25       * @param nearClient        near client instance
26       * @param contractAccountId contract id
27       * @param methodName        view method to call
28       * @param methodType        view or call
29       * @param arguments         contract arguments
30       * @param returnClass       return object class
31       * @param <T>               type
32       * @return a functioncallresult object with type T
33       * @throws Throwable thrown if fails to invoke
34       */
35      <T> FunctionCallResult<T> invoke(NearClient nearClient, String contractAccountId, String methodName,
36                                       ContractMethodType methodType, Base64String arguments,
37                                       Class<T> returnClass) throws Throwable;
38  
39      /**
40       * Proxy invoke the contracts call methods via signed transactions
41       *
42       * @param nearClient        near client instance
43       * @param contractAccountId contract id
44       * @param methodName        view method to call
45       * @param methodType        view or call
46       * @param accountId         user accountId
47       * @param publicKey         user publicKey
48       * @param privateKey        user privateKey to sign the contract
49       * @param arguments         contract arguments
50       * @param deposit           transaction deposit
51       * @return a transactionAwait
52       * @throws Throwable thrown if fails to invoke
53       */
54      TransactionAwait invoke(NearClient nearClient, String contractAccountId, String methodName,
55                              ContractMethodType methodType, String accountId, PublicKey publicKey,
56                              PrivateKey privateKey, ObjectNode arguments, BigInteger deposit) throws Throwable;
57  
58      /**
59       * Proxy invoke the contracts call methods via signed transactions async
60       *
61       * @param nearClient        near client instance
62       * @param contractAccountId contract id
63       * @param methodName        view method to call
64       * @param methodType        view or call
65       * @param accountId         user accountId
66       * @param publicKey         user publicKey
67       * @param privateKey        user privateKey to sign the contract
68       * @param arguments         contract arguments
69       * @param deposit           transaction deposit
70       * @return an EncodedHash of the future transaction
71       * @throws Throwable thrown if fails to invoke
72       */
73      EncodedHash invokeAsync(NearClient nearClient, String contractAccountId, String methodName,
74                              ContractMethodType methodType, String accountId, PublicKey publicKey,
75                              PrivateKey privateKey, ObjectNode arguments, BigInteger deposit) throws Throwable;
76  }
