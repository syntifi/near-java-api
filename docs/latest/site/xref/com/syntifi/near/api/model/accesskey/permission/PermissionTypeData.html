3   import com.syntifi.near.api.exception.NoSuchTypeException;
4   import lombok.Getter;
5   
6   /**
7    * PermissionTypeData
8    *
9    * @author Alexandre Carvalho
10   * @author Andre Bertolace
11   * @since 0.0.1
12   */
13  @Getter
14  public enum PermissionTypeData {
15      FULL_ACCESS("FullAccess", FullAccessPermission.class),
16      FUNCTION_CALL("FunctionCall", FunctionCallPermission.class);
17  
18      private final String name;
19      private final Class<?> clazz;
20  
21      PermissionTypeData(String name, Class<?> clazz) {
22          this.name = name;
23          this.clazz = clazz;
24      }
25  
26      /**
27       * Retrieve Transform implementation class from Transform name
28       *
29       * @param name the name of the permission type
30       * @return the class of given permission type
31       * @throws NoSuchTypeException no such type found
32       */
33      public static Class<?> getClassByName(String name) throws NoSuchTypeException {
34          for (PermissionTypeData t : values()) {
35              if (t.name.equals(name)) {
36                  return t.getClazz();
37              }
38          }
39          throw new NoSuchTypeException(String.format("Permission Type %s invalid/not found.", name));
40      }
41  }
