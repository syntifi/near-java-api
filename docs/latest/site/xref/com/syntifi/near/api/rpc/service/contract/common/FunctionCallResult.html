3   import com.syntifi.crypto.key.encdec.Hex;
4   import com.syntifi.near.api.common.service.NearObjectMapper;
5   import com.syntifi.near.api.rpc.model.contract.ContractFunctionCallResult;
6   import lombok.Getter;
7   import lombok.Setter;
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  
11  import java.io.IOException;
12  
13  /**
14   * A basic typed function call result
15   *
16   * @param <R> the result type to deserialize to
17   * @author Alexandre Carvalho
18   * @author Andre Bertolace
19   * @since 0.2.0
20   */
21  @Getter
22  @Setter
23  public class FunctionCallResult<R> {
24  
25      private static final Logger LOGGER = LoggerFactory.getLogger(FunctionCallResult.class);
26  
27      private ContractFunctionCallResult contractFunctionCallResult;
28      private R result;
29  
30      /**
31       * Creates an instance of FunctionCallResult with the type for deserializing the result data and
32       * the full contract response object
33       *
34       * @param contractFunctionCallResult the full result object from the request
35       * @param clazz                      the class of type R to load the result to
36       * @throws IOException thrown if failed to map json to class
37       */
38      public FunctionCallResult(ContractFunctionCallResult contractFunctionCallResult, Class<R> clazz) throws IOException {
39          this.contractFunctionCallResult = contractFunctionCallResult;
40          if (contractFunctionCallResult.getResult() != null) {
41              LOGGER.debug("Mapping result data to {}", clazz.getSimpleName());
42              LOGGER.trace("Hex string from result data bytes: {}", Hex.encode(contractFunctionCallResult.getResult()));
43              LOGGER.trace("String from result data bytes: {}", new String(contractFunctionCallResult.getResult()));
44              result = NearObjectMapper.INSTANCE.readValue(contractFunctionCallResult.getResult(), clazz);
45          }
46      }
47  }
