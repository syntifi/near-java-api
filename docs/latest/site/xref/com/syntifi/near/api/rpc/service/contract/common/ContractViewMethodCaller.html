12  /**
13   * Caller for contract methods
14   *
15   * @author Alexandre Carvalho
16   * @author Andre Bertolace
17   * @since 0.2.0
18   */
19  @NoArgsConstructor(access = AccessLevel.PRIVATE)
20  public class ContractViewMethodCaller {
21      /**
22       * Calls a contract for specific method
23       *
24       * @param methodName        the method to call
25       * @param returnClass       the type of result object
26       * @param nearClient        the near service to use
27       * @param contractAccountId the contract account id
28       * @param params            the method parameters encoded in Base64
29       * @param <T>               type of the result object
30       * @return a typed FunctionCallResult for the requested contract method
31       * @throws IOException thrown when fails to map json to result object
32       */
33      public static <T> FunctionCallResult<T> call(NearClient nearClient, String contractAccountId, String methodName,
34                                                   Base64String params, Class<T> returnClass) throws IOException {
35          ContractFunctionCallResult contractFunctionCallResult = FunctionCall.builder()
36                  .finality(Finality.OPTIMISTIC)
37                  .methodName(methodName)
38                  .accountId(contractAccountId)
39                  .args(params)
40                  .build()
41                  .call(nearClient);
42          return new FunctionCallResult<>(contractFunctionCallResult, returnClass);
43      }
44  }
