25      // NFT Constants
26      // FT_MINIMUM_STORAGE_BALANCE: nUSDC, nUSDT require minimum 0.0125 NEAR. Came to this conclusion using trial and error.
27      public static String FT_MINIMUM_STORAGE_BALANCE_LARGE = Formats.parseNearAmount("0.0125");
28      // account creation costs 0.00125 NEAR for storage, 0.00000000003 NEAR for gas
29      // https://docs.near.org/docs/api/naj-cookbook#wrap-and-unwrap-near
30      static String FT_MINIMUM_STORAGE_BALANCE = Formats.parseNearAmount("0.00125");
31      static String FT_STORAGE_DEPOSIT_GAS = Formats.parseNearAmount("0.00000000003");
32      // set this to the same value as we use for creating an account and the remainder is refunded
33      static String FT_TRANSFER_GAS = Formats.parseNearAmount("0.00000000003");
34      // contract might require an attached deposit of of at least 1 yoctoNear on transfer methods
35      // "This 1 yoctoNEAR is not enforced by this standard, but is encouraged to do. While ability to receive attached deposit is enforced by this token."
36      // from: https://github.com/near/NEPs/issues/141
37      static int FT_TRANSFER_DEPOSIT = 1;
38  
39      /**
40       * Builds a contract function call and returns a typed result
41       *
42       * @param nearService       the near service instance to use for the contract call
43       * @param contractAccountId the contract's account id
44       * @return a typed function call result
45       * @throws IOException thrown if failed to deserialize result to target class
46       */
47      public static FunctionCallResult<JsonNode> forMetadata(NearService nearService, String contractAccountId) throws IOException {
48          return ContractMethodCaller
49                  .callFor(StakingMethodCaller.ViewMethod.FT_METADATA, JsonNode.class, nearService, contractAccountId, null);
50      }
51  
52      /**
53       * Builds a contract function call and returns a typed result
54       *
55       * @param nearService       the near service instance to use for the contract call
56       * @param contractAccountId the contract's account id
57       * @param accountIdParam    the arguments for the target method
58       * @return a typed function call result
59       * @throws IOException thrown if failed to deserialize result to target class
60       */
61      public static FunctionCallResult<JsonNode> forBalanceOf(NearService nearService, String contractAccountId, AccountIdParam accountIdParam) throws IOException {
62          return ContractMethodCaller
63                  .callFor(StakingMethodCaller.ViewMethod.FT_BALANCE_OF, JsonNode.class, nearService, contractAccountId, accountIdParam);
64      }
65  
66      /**
67       * Builds a contract function call and returns a typed result
68       *
69       * @param nearService       the near service instance to use for the contract call
70       * @param contractAccountId the contract's account id
71       * @param ftTransferParam   the arguments for the target method
72       * @return a typed function call result
73       * @throws IOException thrown if failed to deserialize result to target class
74       */
75      public static FunctionCallResult<JsonNode> forTransferCall(NearService nearService, String contractAccountId, FTTransferParam ftTransferParam) throws IOException {
76          return ContractMethodCaller
77                  .callFor(StakingMethodCaller.ViewMethod.FT_TRANSFER_CALL, JsonNode.class, nearService, contractAccountId, ftTransferParam);
78      }
79  
80      /**
81       * Contract method definitions
82       */
83      public static class StakingMethodCaller extends ContractMethodCaller {
84          @Getter
85          @AllArgsConstructor
86          public enum ViewMethod implements ContractMethod {
87              FT_METADATA("ft_metadata"),
88              FT_TRANSFER_CALL("ft_transfer_call"),
89              FT_BALANCE_OF("ft_balance_of");
90  
91              private final String methodName;
92          }
93      }
