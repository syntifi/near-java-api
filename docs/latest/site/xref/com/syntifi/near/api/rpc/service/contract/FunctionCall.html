15  /**
16   * Base contract function call object
17   *
18   * @author Alexandre Carvalho
19   * @author Andre Bertolace
20   * @since 0.2.0
21   */
22  @Getter
23  @Setter
24  @NoArgsConstructor
25  @AllArgsConstructor
26  @Builder
27  public class FunctionCall {
28      @JsonProperty("account_id")
29      private String accountId;
30  
31      @JsonProperty("method_name")
32      @Setter(AccessLevel.PROTECTED)
33      public String methodName;
34  
35      @JsonProperty("args_base64")
36      private Base64String args;
37  
38      @JsonProperty("finality")
39      private Finality finality;
40  
41      public ContractFunctionCallResult call(NearService nearService) {
42          return nearService.callContractFunction(
43                  this.getFinality(),
44                  this.getAccountId(),
45                  this.getMethodName(),
46                  this.getArgs().getEncodedString());
47      }
48  }
