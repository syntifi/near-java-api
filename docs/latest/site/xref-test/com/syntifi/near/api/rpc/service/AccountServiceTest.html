38      @Test
39      void createImplicitAccountFromMnemonic_should_getStatus_SuccessValueStatus() throws GeneralSecurityException, IOException, MnemonicException.MnemonicLengthException {
40          MnemonicCode mnemonicCode = new MnemonicCode(Language.EN);
41          List<String> words = mnemonicCode.generateSecureRandomWords();
42          PrivateKey newPrivateKey = KeyService.deriveFromSeed(mnemonicCode.toSeed(words, ""));
43          PublicKey newPublicKey = KeyService.derivePublicKey(newPrivateKey);
44          LOGGER.debug("=============> Mnemonic");
45          LOGGER.info(String.join(" ", words));
46          LOGGER.debug("=============> Public key ");
47          LOGGER.debug(Hex.encode(newPublicKey.getData()));
48          LOGGER.debug("=============> Privatekey ");
49          LOGGER.debug(Hex.encode(newPrivateKey.getData()));
50          String signerId = "syntifi-bob.testnet";
51          BigInteger amount = new BigInteger(Formats.parseNearAmount("1"), 10);
52          PrivateKey privateKey = bobNearPrivateKey;
53          PublicKey publicKey = bobNearPublicKey;
54          TransactionAwait transactionAwait = TransferService
55                  .sendTransferActionAwait(nearClient, signerId, Hex.encode(newPublicKey.getData()),
56                          publicKey, privateKey, amount);
57          assertInstanceOf(Status.class, transactionAwait.getStatus());
58      }
