31      @Test
32      void createImplicitAccountFromMnemonic_should_getStatus_SuccessValueStatus() throws GeneralSecurityException, IOException, MnemonicException.MnemonicLengthException {
33          MnemonicCode mnemonicCode = new MnemonicCode(Language.EN);
34          List<String> words = mnemonicCode.generateSecureRandomWords();
35          PrivateKey newPrivateKey = KeyService.deriveFromSeed(mnemonicCode.toSeed(words, ""));
36          PublicKey newPublicKey = KeyService.derivePublicKey(newPrivateKey);
37          LOGGER.debug("=============> Mnemonic");
38          LOGGER.info(String.join(" ", words));
39          LOGGER.debug("=============> Public key ");
40          LOGGER.debug(Hex.encode(newPublicKey.getData()));
41          LOGGER.debug("=============> Privatekey ");
42          LOGGER.debug(Hex.encode(newPrivateKey.getData()));
43          String signerId = "syntifi-alice.testnet";
44          BigInteger amount = new BigInteger(Formats.parseNearAmount("1"), 10);
45          PrivateKey privateKey = aliceNearPrivateKey;
46          PublicKey publicKey = aliceNearPublicKey;
47          TransactionAwait transactionAwait = TransactionService
48                  .sendTransferActionAwait(nearService, signerId, Hex.encode(newPublicKey.getData()),
49                          publicKey, privateKey, amount);
50          assertInstanceOf(SuccessValueStatus.class, transactionAwait.getStatus());
51      }

55          Random rnd = new Random();

57          String newAccountId = Math.abs(rnd.nextInt()) + "." + signerId;
58          BigInteger amount = new BigInteger(Formats.parseNearAmount("1"), 10);
59          PrivateKey privateKey = aliceNearPrivateKey;
60          PublicKey publicKey = aliceNearPublicKey;
61  
62          EncodedHash transactionHash = AccountService
63                  .createSubAccountAsync(nearService, newAccountId, amount, signerId, publicKey, privateKey);
64          assertTrue(transactionHash.getDecodedHash().length > 0);
65      }
