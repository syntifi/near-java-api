1   package com.syntifi.near.api.service.exception;
2   
3   import com.fasterxml.jackson.core.JsonProcessingException;
4   import com.fasterxml.jackson.databind.JsonNode;
5   import com.fasterxml.jackson.databind.ObjectMapper;
6   import com.fasterxml.jackson.databind.node.ObjectNode;
7   import com.googlecode.jsonrpc4j.ExceptionResolver;
8   import com.syntifi.near.api.service.NearObjectMapper;
9   
10  /**
11   * Custom exception handler for jsonrpc4j client
12   *
13   * @author Alexandre Carvalho
14   * @author Andre Bertolace
15   * @since 0.0.1
16   */
17  public class NearServiceExceptionResolver implements ExceptionResolver {
18      private static final ObjectMapper objectMapper = new NearObjectMapper();
19  
20      @Override
21      public Throwable resolveException(ObjectNode response) {
22          try {
23              JsonNode errorNode = response.get("error");
24              NearServiceErrorData error = objectMapper.treeToValue(errorNode, NearServiceErrorData.class);
25              return new NearServiceException(error);
26          } catch (JsonProcessingException | IllegalArgumentException e) {
27              return new NearServiceException(String.format("Could not extract error, response was: %s", response), e);
28          }
29      }
30  }
