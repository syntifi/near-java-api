32      /**
33       * Sends a {@link TransferAction} transaction waiting for result using  {@link NearService#sendTransactionAwait(String)}
34       *
35       * @param nearService      the near service instance to use
36       * @param signerId         the signer id
37       * @param receiverId       the receiver id
38       * @param signerPublicKey  signer public key
39       * @param signerPrivateKey signer private key
40       * @param amount           the amount to transfer
41       * @return {@link TransactionAwait} object with the result
42       * @throws GeneralSecurityException thrown if failed to sign the transaction
43       */
44      public static TransactionAwait sendTransferActionAwait(NearService nearService, String signerId, String receiverId,
45                                                             PublicKey signerPublicKey, PrivateKey signerPrivateKey,
46                                                             BigInteger amount)
47              throws GeneralSecurityException {
48          return sendTransactionAwait(nearService, signerId, receiverId, signerPublicKey, signerPrivateKey,
49                  Arrays.asList(
50                          TransferAction.builder()
51                                  .deposit(amount)
52                                  .build()));
53      }

78      /**
79       * Sends a list of {@link Action} transaction waiting for result using {@link NearService#sendTransactionAwait(String)}
80       *
81       * @param nearService      the near service instance to use
82       * @param signerId         the signer id
83       * @param receiverId       the receiver id
84       * @param signerPublicKey  signer public key
85       * @param signerPrivateKey signer private key
86       * @param actionList       list of {@link Action} to send
87       * @return {@link TransactionAwait} object with the result
88       * @throws GeneralSecurityException thrown if failed to sign the transaction
89       */
90      public static TransactionAwait sendTransactionAwait(NearService nearService, String signerId, String receiverId,
91                                                          PublicKey signerPublicKey, PrivateKey signerPrivateKey,
92                                                          List<Action> actionList)
93              throws GeneralSecurityException {
94          return nearService.sendTransactionAwait(prepareTransactionForActionList(nearService, signerId, receiverId, signerPublicKey, signerPrivateKey, actionList));
95      }

97      /**
98       * Sends a list of {@link Action} transaction waiting for result using  {@link NearService#sendTransactionAsync(String)}
99       *
100      * @param nearService      the near service instance to use
101      * @param signerId         the signer id
102      * @param receiverId       the receiver id
103      * @param signerPublicKey  signer public key
104      * @param signerPrivateKey signer private key
105      * @param actionList       list of {@link Action} to send
106      * @return {@link TransactionAwait} object with the result
107      * @throws GeneralSecurityException thrown if failed to sign the transaction
108      */
109     public static EncodedHash sendTransactionAsync(NearService nearService, String signerId, String receiverId,
110                                                    PublicKey signerPublicKey, PrivateKey signerPrivateKey,
111                                                    List<Action> actionList)
112             throws GeneralSecurityException {
113         return EncodedHash.builder()
114                 .encodedHash(nearService.sendTransactionAsync(
115                         prepareTransactionForActionList(nearService, signerId, receiverId, signerPublicKey, signerPrivateKey, actionList)))
116                 .build();
117     }

119     /**
120      * Prepares the transaction to send
121      *
122      * @param nearService      the near service instance to use
123      * @param signerId         the signer id
124      * @param receiverId       the receiver id
125      * @param signerPublicKey  signer public key
126      * @param signerPrivateKey signer private key
127      * @param actionList       list of {@link Action} to send
128      * @return the base64 encoded signed transaction string
129      * @throws GeneralSecurityException thrown if failed to sign the transaction
130      */
131     private static String prepareTransactionForActionList(NearService nearService, String signerId, String receiverId,
132                                                           PublicKey signerPublicKey, PrivateKey signerPrivateKey,
133                                                           List<Action> actionList)
134             throws GeneralSecurityException {
135         Block block = nearService.getBlock(Finality.FINAL);
136 
137         AccessKey accessKey = nearService.viewAccessKey(Finality.FINAL, signerId, signerPublicKey.toEncodedBase58String());
138 
139         long nextNonce = accessKey.getNonce() + 1L;
140 
141         Transaction transaction = Transaction
142                 .builder()
143                 .signerId(signerId)
144                 .publicKey(signerPublicKey)
145                 .nonce(nextNonce)
146                 .receiverId(receiverId)
147                 .blockHash(block.getHeader().getHash().getDecodedHash())
148                 .actions(actionList)
149                 .build();
150 
151         byte[] serializedTx = Borsh.serialize(transaction);
152         byte[] hashedTx = Sha256.digest(serializedTx);
153         byte[] signedTx = signerPrivateKey.getPrivateKey().sign(hashedTx);
154 
155         SignedTransaction signedTransaction =
156                 SignedTransaction.builder()
157                         .transaction(transaction)
158                         .signature(Signature.builder()
159                                 .keyType(signerPublicKey.getKeyType())
160                                 .data(signedTx).build())
161                         .build();
162 
163         byte[] borshTx = Borsh.serialize(signedTransaction);
164 
165         return Base64.getEncoder().encodeToString(borshTx);
166     }
