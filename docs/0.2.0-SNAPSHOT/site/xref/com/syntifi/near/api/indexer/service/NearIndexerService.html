1   package com.syntifi.near.api.indexer.service;
2   
3   import com.syntifi.near.api.common.service.NearObjectMapper;
4   import com.syntifi.near.api.indexer.model.AccountIdList;
5   import com.syntifi.near.api.indexer.model.NearValue;
6   import com.syntifi.near.api.indexer.model.RecentActivity;
7   import com.syntifi.near.api.indexer.model.StakingDeposit;
8   import okhttp3.Headers;
9   import okhttp3.OkHttpClient;
10  import retrofit2.Call;
11  import retrofit2.Retrofit;
12  import retrofit2.converter.jackson.JacksonConverterFactory;
13  import retrofit2.http.GET;
14  import retrofit2.http.Path;
15  
16  import java.util.List;
17  
18  /**
19   * Near indexer service uses http API to retrieve useful information on chain data
20   *
21   * @author Alexandre Carvalho
22   * @author Andre Bertolace
23   * @since 0.2.0
24   */
25  public interface NearIndexerService {
26  
27      /**
28       * Fetches all likely NFTs given an accountId
29       *
30       * @param accountId the account to search information for
31       * @return a retrofit call for a AccountIdList
32       */
33      @GET("account/{accountId}/likelyNFTs")
34      Call<AccountIdList> getAccountLikelyNFTs(@Path("accountId") String accountId);
35  
36      /**
37       * Fetches all likely FTs (Tokens) given an accountId
38       *
39       * @param accountId the account to search information for
40       * @return a retrofit call for a AccountIdList
41       */
42      @GET("account/{accountId}/likelyTokens")
43      Call<AccountIdList> getAccountLikelyFTs(@Path("accountId") String accountId);
44  
45      /**
46       * Fetches staing deposits for an account
47       *
48       * @param accountId the account to search information for
49       * @return a retrofit call for a list of StakingDeposits
50       */
51      @GET("staking-deposits/{accountId}")
52      Call<List<StakingDeposit>> getStakingDeposits(@Path("accountId") String accountId);
53  
54      /**
55       * Fetches a list of recent activities for one account
56       *
57       * @param accountId the account to fetch activity
58       * @return list of recent activity for the account
59       */
60      @GET("account/{accountId}/activity")
61      Call<RecentActivity> getNearRecentActivity(@Path("accountId") String accountId);
62  
63      /**
64       * Fetches the value of near in fiat currency
65       *
66       * @return the Near fiat value
67       */
68      @GET("/fiat")
69      Call<NearValue> getNearValue();
70  
71      /**
72       * Fetches a list of accounts for a given public key
73       *
74       * @param publicKey the account public key
75       * @return list containing all account ids
76       */
77      @GET("publicKey/{publicKey}/accounts")
78      Call<List<String>> getNearAccounts(@Path("publicKey") String publicKey);
79  
80      /**
81       * NearIndexerService builder
82       *
83       * @param url the indexer url to connect to
84       * @return the indexer service instance
85       */
86      static NearIndexerService usingPeer(String url) {
87          Headers customHeaders = new Headers.Builder()
88                  .add("Content-Type", "application/json")
89                  .add("Cache-Control", "no-cache")
90                  .build();
91  
92          Retrofit retrofit = new Retrofit.Builder()
93                  .client(new OkHttpClient.Builder()
94                          .addInterceptor(
95                                  chain -> chain.proceed(chain.request().newBuilder().headers(customHeaders).build())).build())
96                  .baseUrl("https://" + url)
97                  .addConverterFactory(JacksonConverterFactory.create(NearObjectMapper.INSTANCE))
98                  .build();
99  
100         return retrofit.create(NearIndexerService.class);
101     }
102 }
