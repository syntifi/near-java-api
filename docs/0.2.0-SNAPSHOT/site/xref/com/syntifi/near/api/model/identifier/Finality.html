5   /**
6    * The finality param has two options: optimistic and final.
7    * - optimistic uses the latest block recorded on the node that responded to
8    * your query (&lt;1 second delay after the transaction is submitted)
9    * - final is for a block that has been validated on at least 66% of the nodes
10   * in the network (usually takes 2 blocks / approx. 2 second delay)
11   * 
12   * @author Alexandre Carvalho
13   * @author Andre Bertolace
14   * @since 0.0.1
15   */
16  public enum Finality {
17      OPTIMISTIC("optimistic"), FINAL("final");
18  
19      private final String type;
20  
21      Finality(String type) {
22          this.type = type;
23      }
24  
25      @Override
26      @JsonValue
27      public String toString() {
28          return type;
29      }
30  }
